    "# Lesson 3: Stake effects and presentation order — de Hollander et al. (2024)\n",
    "\n",
    "## Background\n",
    "\n",
    "De Hollander et al. (2024, *Nature Human Behaviour*) tested whether perceptual noise\n",
    "during the encoding of numerical magnitudes explains risk aversion and its interaction\n",
    "with presentation order.  The key design feature: the **order** in which the safe and\n",
    "risky options are presented is randomised across trials, allowing the model to disentangle\n",
    "$\\nu_1$ (noise on the first-presented option) from $\\nu_2$ (noise on the second-presented\n",
    "option).\n",
    "\n",
    "This produces a distinctive **presentation-order × stake-size interaction**: when the\n",
    "safe option comes first, high safe stakes are compressed downward more strongly by the\n",
    "prior → safe looks less attractive → the observer becomes more risk-seeking for high\n",
    "stakes.  When the risky option comes first, the same mechanism operates on the risky\n",
    "stakes → risk aversion for high stakes.\n",
    "\n",
    "Standard models (EU, KLW with a shared noise) **cannot** capture this asymmetry.\n",
    "\n",
    "### Models compared\n",
    "\n",
    "| Model | Class | Key parameters |\n",
    "|-------|-------|----------------|\n",
    "| Expected Utility (EU) | `ExpectedUtilityRiskModel` | `alpha`, `sigma` |\n",
    "| KLW | `RiskModel(prior_estimate='klw', fit_seperate_evidence_sd=False)` | `evidence_sd`, `prior_sd` (shared) |\n",
    "| PMCM | `RiskModel(prior_estimate='full', fit_seperate_evidence_sd=True)` | `n1_evidence_sd`, `n2_evidence_sd`, `risky/safe_prior_mu/sd` |\n",
    "\n",
    "We fit all three on both the **dot-cloud** (fMRI sessions 3T+7T) and the **symbolic**\n",
    "(Arabic numerals) datasets, then compare posterior predictives against the interaction\n",
    "pattern."
   ]

    "def prep_df(df):\n",
    "    \"\"\"Add log_ratio, chose_risky, n_safe, order flag, and binned columns.\"\"\"\n",
    "    df = df.reset_index()\n",
    "    risky_first = df['p1'] == 0.55\n",
    "    df['log_ratio']     = np.log(\n",
    "        np.where(risky_first, df['n1'], df['n2']) /\n",
    "        np.where(risky_first, df['n2'], df['n1']))\n",
    "    df['chose_risky']   = np.where(risky_first, ~df['choice'], df['choice'])\n",
    "    df['n_safe']        = np.where(risky_first, df['n2'], df['n1'])\n",
    "    df['risky_first']   = risky_first\n",
    "    df['order']         = np.where(risky_first, 'Risky first', 'Safe first')\n",
    "    df['log_ratio_bin'] = (pd.cut(df['log_ratio'], bins=10)\n",
    "                             .apply(lambda x: x.mid).astype(float))\n",
    "    df['n_safe_bin']    = pd.qcut(df['n_safe'], q=3, labels=['Low stakes', 'Mid stakes', 'High stakes'])\n",
    "    return df\n",
    "\n",
    "df_dot_p = prep_df(df_dot)\n",
    "df_sym_p = prep_df(df_sym)"
   ]

    "stake_pal = {'Low stakes': '#4C72B0', 'Mid stakes': '#DD8452', 'High stakes': '#55A868'}\n",
    "\n",
    "def plot_interaction(df_p, axes_row, task_label):\n",
    "    \"\"\"Plot stake x order interaction, one axis per order condition.\"\"\"\n",
    "    for ax, order_val in zip(axes_row, ['Risky first', 'Safe first']):\n",
    "        sub  = df_p[df_p['order'] == order_val]\n",
    "        subj = (sub.groupby(['subject', 'log_ratio_bin', 'n_safe_bin'])['chose_risky']\n",
    "                   .mean().reset_index())\n",
    "        subj = subj[subj.groupby(['log_ratio_bin', 'n_safe_bin'])\n",
    "                        ['subject'].transform('count') >= 3]\n",
    "        hue_order = ['Low stakes', 'Mid stakes', 'High stakes']\n",
    "        sns.lineplot(data=subj, x='log_ratio_bin', y='chose_risky',\n",
    "                     hue='n_safe_bin', style='n_safe_bin',\n",
    "                     hue_order=hue_order, style_order=hue_order,\n",
    "                     palette=stake_pal, markers=True, dashes=False,\n",
    "                     errorbar='se', ax=ax)\n",
    "        ax.axhline(.5,            ls='--', c='gray', lw=1)\n",

    "        ax.set_ylim(-.05, 1.05)\n",
    "        ax.set_xlabel('log(risky / safe)')\n",

    "def add_model_ppc(df_orig, df_prepped, model, idata, model_name, n_ppc_samples=50):\n",
    "    \"\"\"Return df_prepped with added posterior-predictive columns.\"\"\"\n",
    "    ppc_df = model.ppc(df_orig, idata, var_names=['p'])\n",
    "    ppc_p  = ppc_df.xs('p', level='variable')\n",
    "    cols   = np.random.choice(ppc_p.columns,\n",
    "                               size=min(n_ppc_samples, ppc_p.shape[1]),\n",
    "                               replace=False)\n",
    "    ppc_p  = ppc_p[cols]\n",
    "    risky_first = df_prepped['risky_first'].values\n",
    "    p_risky = np.where(risky_first[:, None], 1 - ppc_p.values, ppc_p.values)\n",
    "    df_out = df_prepped.copy()\n",
    "    df_out['p_mean'] = p_risky.mean(1)\n",
    "    df_out['p_lo']   = np.percentile(p_risky, 2.5,  axis=1)\n",
    "    df_out['p_hi']   = np.percentile(p_risky, 97.5, axis=1)\n",
    "    df_out['model']  = model_name\n",
    "    return df_out\n",
    "\n",
    "\n",
    "def plot_ppc_interaction(df_pred, model_name, axes_row):\n",
    "    hue_order = ['Low stakes', 'Mid stakes', 'High stakes']\n",
    "    for ax, order_val in zip(axes_row, ['Risky first', 'Safe first']):\n",
    "        sub  = df_pred[df_pred['order'] == order_val]\n",
    "        obs  = sub.groupby(['n_safe_bin', 'log_ratio_bin'])['chose_risky'].mean().reset_index()\n",
    "        pred = (sub.groupby(['n_safe_bin', 'log_ratio_bin'])[['p_mean', 'p_lo', 'p_hi']]\n",
    "                   .mean().reset_index())\n",
    "        for sbin in hue_order:\n",
    "            o = obs[obs['n_safe_bin']  == sbin]\n",
    "            p = pred[pred['n_safe_bin'] == sbin]\n",
    "            if len(o) == 0:\n",
    "                continue\n",
    "            c = stake_pal[sbin]\n",
    "            ax.fill_between(p['log_ratio_bin'], p['p_lo'], p['p_hi'],\n",
    "                            color=c, alpha=.20)\n",
    "            ax.plot(p['log_ratio_bin'], p['p_mean'], color=c, lw=2, label=sbin)\n",
    "            ax.scatter(o['log_ratio_bin'], o['chose_risky'],\n",
    "                       color=c, s=25, zorder=5, alpha=.85)\n",
    "        ax.axhline(.5,            ls='--', c='gray', lw=1)\n",
    "        ax.axvline(np.log(1/.55), ls='--', c='#333333', lw=1.5)\n",
    "        ax.set_ylim(-.05, 1.05)\n",
    "        ax.set_title(f'{model_name} — {order_val}', fontsize=9)\n",
    "        ax.set_xlabel('log(risky / safe)')\n",
    "        ax.set_ylabel('P(chose risky)')\n",
    "        ax.legend(title='Safe stake', fontsize=7, loc='upper left')\n",
    "        sns.despine(ax=ax)\n",
    "\n",
    "\n",
    "fig, axes = plt.subplots(3, 2, figsize=(12, 13), sharey=True)\n",
    "for (mdl, idat, name), row in zip(\n",
    "        [(model_eu,   idata_eu,   'EU'),\n",
    "         (model_klw,  idata_klw,  'KLW'),\n",
    "         (model_full, idata_full, 'PMCM')],\n",
    "        axes):\n",
    "    df_pred = add_model_ppc(df_dot, df_dot_p, mdl, idat, name)\n",
    "    plot_ppc_interaction(df_pred, name, row)\n",
    "\n",
    "plt.suptitle('Posterior predictive checks — dot-cloud data  (dots = observed, shading = 95 % CI)',\n",
    "             fontsize=11, y=1.01)\n",
    "plt.tight_layout()"
   ]
